
##禁用自动旋转

```
- (BOOL)shouldAutorotate
{
    return NO;
}
```

##在要变换的地方

```
[[UIApplication sharedApplication] 
setStatusBarOrientation:UIInterfaceOrientationLandscapeRight animated:YES];

flat duration = UIApplication sharedApplication].statusBarOrientationAnimationDuration;

[UIView animateWithDuration:duration animations: ^{
            CGAffineTransform transform = <#你的变换#>
            <#相关要旋转的view#>.transform = transform;
        }];
        
```

##恢复变换

```
[[UIApplication sharedApplication] setStatusBarOrientation:UIInterfaceOrientationPortrait animated:YES];
        
[UIView animateWithDuration:[UIApplication sharedApplication].statusBarOrientationAnimationDuration animations:^{
            _playerView.transform = CGAffineTransformIdentity;
        }];

```
